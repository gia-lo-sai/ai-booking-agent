📱 SETUP RAPIDO SU TERMUX
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣  Inizializza progetto:
    bash scripts/init.sh

2️⃣  Oppure step-by-step:
    bash scripts/setup-hcloud.sh
    cp .env.example .env
    nano .env  # Aggiungi HCLOUD_TOKEN e ANTHROPIC_API_KEY

3️⃣  Deploy su hcloud:
    npm run deploy

4️⃣  Accedi all'app:
    http://YOUR_SERVER_IP

💡 TIPS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

• Usa sempre "ai-booking-agent" come nome progetto
• Il server type cx11 costa ~€4/mese su hcloud
• Backup automatico: npm run hcloud:backup
• Per sviluppo locale: npm run dev
• Logs in tempo reale: npm run hcloud:logs


🆘 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

hcloud non trovato:
    npm install -g @hetznercloud/cli

Errore deploy:
    bash scripts/check-status.sh
    Verifica .env

Database error:
    npm run hcloud:ssh
    cd /opt/ai-booking-agent
    docker-compose logs postgres
